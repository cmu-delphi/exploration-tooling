align_version_to_wednesday <- function(dt) {
  dt |>
    as_tibble() |>
    mutate(
      version_wed = lubridate::floor_date(version, "week", week_start = 3L),
      is_wednesday = lubridate::wday(version, week_start = 1L) == 3L
    ) |>
    group_by(geo_value, time_value, version_wed) |>
    arrange(desc(is_wednesday)) |>
    slice(1) |>
    ungroup() |>
    mutate(version = version_wed) |>
    select(-version_wed, -is_wednesday)
}

Data Transformation

The archive stores value as the revised count for a reference week at each version date. baselinenowcast expects a reporting triangle where each cell is the incremental count added at each delay step. We compute that as the first-difference within each (geo, time_value) group.

Code 
#' Align archive versions onto Wednesdays using ceiling_date.
#'
#' Within each (geo_value, time_value, ceiled version) group, prefers the row
#' whose version was already a Wednesday; otherwise keeps the first.

#' Convert a subset of the NHSN epi_archive to reporting triangle long format.
#'
#' Aligns versions to Wednesdays, filters to a single geo and all versions up
#' to nowcast_date, then computes incremental counts per reference week.
incremental_data <- nhsn_archive$DT |>
  filter(time_value >= min(version)) |>
  group_by(geo_value, time_value) |>
  arrange(geo_value, time_value, version) |>
  mutate(count = as.integer(value - lag(value, default = 0L))) |>
  ungroup() |>
  rename(reference_date = time_value, report_date = version) |>
  select(geo_value, reference_date, report_date, count)
make_reporting_long <- function(archive_dt, geo, nowcast_date) {
  archive_dt |>
    filter(geo_value == geo, version <= nowcast_date) |>
    filter(time_value >= min(version)) |>
    arrange(time_value, version) |>
    group_by(time_value) |>
    mutate(count = as.integer(value - lag(value, default = 0L))) |>
    ungroup() |>
    rename(reference_date = time_value, report_date = version) |>
    select(reference_date, report_date, count)
}

nowcast_date <- as.Date("2026-02-04")
geo <- "ak"
run_retro_nowcast <- function(nowcast_date, incremental_data, geo,
                              truncate_quantile = 0.5, draws = 500L, eval_window_weeks = 7L) {
  long_data <- incremental_data |>
    filter(geo_value == geo, report_date <= nowcast_date) |>
    select(-geo_value)

  has_negatives <- any(long_data$count < 0)

  rep_tri_trunc <- tryCatch(
    suppressMessages(
      as_reporting_triangle(long_data, delays_unit = "weeks") |>
        truncate_to_quantile(truncate_quantile)
    ),
    error = function(e) {NULL}
  )
  if (is.null(rep_tri_trunc) || nrow(as.data.frame(rep_tri_trunc)) == 0) return(NULL)

  set.seed(as.integer(nowcast_date))
  draws_df <- tryCatch(
    baselinenowcast(rep_tri_trunc, scale_factor = 3, prop_delay = 0.5, draws = draws),
    error = function(e) {NULL})
  if (is.null(draws_df)) return(NULL)

  draws_df |>
    filter(nowcast_date - reference_date < eval_window_weeks * 7) |>
    group_by(reference_date) |>
    summarise(
      median = median(pred_count),
      q05 = quantile(pred_count, 0.05),
      q25 = quantile(pred_count, 0.25),
      q75 = quantile(pred_count, 0.75),
      q95 = quantile(pred_count, 0.95)
    ) |>
    mutate(nowcast_date = nowcast_date, geo_value = geo, has_negatives = has_negatives)
}

calc_naive_wis <- function(as_of_date) {
  nhsn_archive |>
    epix_as_of(as_of_date) |>
    rename(reference_date = time_value, as_of_value = value) |>
    left_join(final_values_retro, by = join_by(geo_value, reference_date)) |>
    mutate(nowcast_date = as_of_date, naive_wis = abs(as_of_value - final_value)) |>
    select(geo_value, nowcast_date, reference_date, naive_wis) |>
    filter(naive_wis > 0)
}
# Naive WIS = absolute error of the as-of value (WIS of a point forecast)
naive_wis <- map(retro_nowcasts$nowcast_date |> unique(), calc_naive_wis) |> list_rbind()

scores <- retro_nowcasts |>
  left_join(final_values_retro, by = join_by(geo_value, reference_date)) |>
  filter(!is.na(final_value)) |>
  pivot_longer(
    cols = c(q05, q25, median, q75, q95),
    names_to = "quantile_level",
    values_to = "predicted"
  ) |>
  mutate(
    quantile_level = case_match(
      quantile_level,
      "q05" ~ 0.05, "q25" ~ 0.25, "median" ~ 0.5, "q75" ~ 0.75, "q95" ~ 0.95
