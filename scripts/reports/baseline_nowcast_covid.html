align_version_to_wednesday <- function(dt) {
  dt |>
    as_tibble() |>
    mutate(
      version_wed = lubridate::floor_date(version, "week", week_start = 3L),
      is_wednesday = lubridate::wday(version, week_start = 1L) == 3L
    ) |>
    group_by(geo_value, time_value, version_wed) |>
    arrange(desc(is_wednesday)) |>
    slice(1) |>
    ungroup() |>
    mutate(version = version_wed) |>
    select(-version_wed, -is_wednesday)
}

Data Transformation

The archive stores value as the revised count for a reference week at each version date. baselinenowcast expects a reporting triangle where each cell is the incremental count added at each delay step. We compute that as the first-difference within each (geo, time_value) group.

Code 
#' Align archive versions onto Wednesdays using ceiling_date.
#'
#' Within each (geo_value, time_value, ceiled version) group, prefers the row
#' whose version was already a Wednesday; otherwise keeps the first.

#' Convert a subset of the NHSN epi_archive to reporting triangle long format.
#'
#' Aligns versions to Wednesdays, filters to a single geo and all versions up
#' to nowcast_date, then computes incremental counts per reference week.
incremental_data <- nhsn_archive$DT |>
  filter(time_value >= min(version)) |>
  group_by(geo_value, time_value) |>
  arrange(geo_value, time_value, version) |>
  mutate(count = as.integer(value - lag(value, default = 0L))) |>
  ungroup() |>
  rename(reference_date = time_value, report_date = version) |>
  select(geo_value, reference_date, report_date, count)
make_reporting_long <- function(archive_dt, geo, nowcast_date) {
  archive_dt |>
    filter(geo_value == geo, version <= nowcast_date) |>
    filter(time_value >= min(version)) |>
    arrange(time_value, version) |>
    group_by(time_value) |>
    mutate(count = as.integer(value - lag(value, default = 0L))) |>
    ungroup() |>
    rename(reference_date = time_value, report_date = version) |>
    select(reference_date, report_date, count)
}

Retrospective Evaluation

To evaluate how well baselinenowcast would have performed in practice, we re-run it at every weekly version date in the archive, for every state/territory. Each call receives only the data that would have been available at that point in time — no future information leaks in.

We don’t nowcast during the first 10 weeks of the archive to ensure the model has enough historical data to estimate a stable delay distribution before we start scoring. run_retro_nowcast wraps the nowcasting pipeline in tryCatch so that individual failures (e.g. states with too few reports at early dates) are silently skipped rather than aborting the whole run. The has_negatives flag records whether any incremental counts were negative for that (geo, nowcast_date) combination — these arise from downward revisions and are clipped to zero before being passed to the model.

Results are restricted to the 7 reference weeks closest to each nowcast date (eval_window_weeks = 7), which covers the range where real-time revision uncertainty is highest.

This next block is the bulk of the retro forecasting code; run_retro_nowcast fits baselinenowcast on a single nowcast_date and geo_value. Note that this is using parallelism via furrr::future_pmap.

Code 
run_retro_nowcast <- function(nowcast_date, incremental_data, geo,
                              truncate_quantile = 0.99, draws = 500L, eval_window_weeks = 7L) {
  long_data <- incremental_data |>
    filter(geo_value == geo, report_date <= nowcast_date) |>
    select(-geo_value)

  has_negatives <- any(long_data$count < 0)

  rep_tri_trunc <- tryCatch(
    suppressMessages(
      as_reporting_triangle(long_data, delays_unit = "weeks") |>
        truncate_to_quantile(truncate_quantile)
    ),
    error = function(e) {NULL}
  )
  if (is.null(rep_tri_trunc) || nrow(as.data.frame(rep_tri_trunc)) == 0) return(NULL)

  set.seed(as.integer(nowcast_date))
  draws_df <- tryCatch(
    baselinenowcast(rep_tri_trunc, scale_factor = 3, prop_delay = 0.5, draws = draws),
    error = function(e) {NULL})
  if (is.null(draws_df)) return(NULL)

  draws_df |>
    filter(nowcast_date - reference_date < eval_window_weeks * 7) |>
    group_by(reference_date) |>
    summarise(
      median = median(pred_count),
      q05 = quantile(pred_count, 0.05),
      q25 = quantile(pred_count, 0.25),
      q75 = quantile(pred_count, 0.75),
      q95 = quantile(pred_count, 0.95)
    ) |>
    mutate(nowcast_date = nowcast_date, geo_value = geo, has_negatives = has_negatives)
}

WIS Scoring

Nowcasts are scored against the most recent available version of each reference week’s count, which we treat as ground truth. Scores are computed using scoringutils::score() on the five quantile levels (5th, 25th, 50th, 75th, 95th) returned by the nowcaster.

The naive baseline is the absolute error of the as-of value - i.e. what you’d get by simply reporting the current value without any nowcast adjustment. Cases where the naive baseline is NA (the as-of snapshot had no value for that reference date) are dropped. Cases where the naive WIS would be zero — meaning the as-of value already matched the final value exactly — are retained, since excluding them would make the nowcaster look artificially better.

Code 
final_values_retro <- nhsn_archive |>
  epix_as_of_current() |>
  select(geo_value, reference_date = time_value, final_value = value)

calc_naive_wis <- function(as_of_date) {
  nhsn_archive |>
    epix_as_of(as_of_date) |>
    rename(reference_date = time_value, as_of_value = value) |>
    left_join(final_values_retro, by = join_by(geo_value, reference_date)) |>
    mutate(nowcast_date = as_of_date, naive_wis = abs(as_of_value - final_value)) |>
    select(geo_value, nowcast_date, reference_date, as_of_value, naive_wis)
}

Skill Score

Skill score is 1 − mean(WIS) / mean(naive WIS): ranging from -1 to 1, positive means the nowcaster is better the naive as-of baseline on average, negative means worse. The naive baseline is zero-WIS when the as-of value already equals the final value — cases where the nowcaster can only match perfection, not exceed it. The orange diamonds show what fraction of cases in each horizon bin have a zero-WIS naive baseline; high fractions mechanically suppress the skill score.

rank_pairs <- function(rankings, metric_col, n, descending = TRUE, min_gap_weeks = 8) {
  ordered <- rankings |>
    arrange(if (descending) desc(!!sym(metric_col)) else !!sym(metric_col))

  selected <- vector("list", n)
  selected_dates <- list()
  count <- 0L

  for (ii in seq_len(nrow(ordered))) {
    row <- ordered[ii, ]
    geo <- row$geo_value
    date <- row$nowcast_date
    existing <- selected_dates[[geo]]
    too_close <- !is.null(existing) &&
      any(abs(as.numeric(difftime(date, existing, units = "weeks"))) < min_gap_weeks)
    if (!too_close) {
      count <- count + 1L
      selected[[count]] <- row
      selected_dates[[geo]] <- if (is.null(existing)) date else c(existing, date)
    }
    if (count >= n) break
  }

  bind_rows(selected[seq_len(count)]) |>
    mutate(panel_label = factor(
      paste(geo_value, nowcast_date),
      levels = paste(geo_value, nowcast_date)
    ))
}
